def min_fee(pages_to_print):
    # 여기에 코드를 작성하세요
    sort_fee = sorted(pages_to_print)

    total_fee = 0

    for i in range(len(pages_to_print)):
        total_fee += sort_fee[i] * len(sort_fee[i:])

    return total_fee

# 테스트 코드
print(min_fee([6, 11, 4, 1]))
print(min_fee([3, 2, 1]))
print(min_fee([3, 1, 4, 3, 2]))
print(min_fee([8, 4, 2, 3, 9, 23, 6, 8]))
