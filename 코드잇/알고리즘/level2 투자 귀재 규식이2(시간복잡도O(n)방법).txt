def sublist_max(profits):
    # 여기에 코드를 작성하세요
    max_value= 0
    before_value= 0
    for i in profits:
        after_value = before_value + i
        if before_value < after_value or after_value > 0:
            before_value = after_value
        else:
            before_value = 0
        max_value= max(max_value,before_value)
    return max_value

# 테스트 코드
print(sublist_max([7, -3, 4, -8]))
print(sublist_max([-2, -3, 4, -1, -2, 1, 5, -3, -1]))
