def insertion_sort(a):
    for i in range(len(a)-1):
        index=i+1
        while  0 <= i:
            if a[index] < a[i]:
                a[index],a[i] = a[i],a[index]
                index=i
            i -=1
               
            
    return a


a=[99, 205, 7, 35, 9, 3]
print(insertion_sort(a))
